// 関数宣言 function greet(name) { return "こんにちは、" + name + "さん！"; } // 関数式 const add = function(a, b) { return a + b; }; // アロー関数 const multiply = (x, y) => x * y; // 関数の呼び出し console.log(greet("田中")); console.log(add(5, 3)); console.log(multiply(4, 6));

配列とメソッド
データの集合を効率的に操作する方法を学習します。
// 配列の作成 const fruits = ["apple", "banana", "orange"]; const numbers = new Array(1, 2, 3, 4, 5); // 基本的なメソッド fruits.push("grape"); // 末尾に追加 fruits.pop(); // 末尾から削除 fruits.unshift("cherry"); // 先頭に追加 fruits.shift(); // 先頭から削除 // 高次関数（配列メソッド） const doubled = numbers.map(n => n * 2); const evens = numbers.filter(n => n % 2 === 0); const sum = numbers.reduce((acc, n) => acc + n, 0); // 検索メソッド const found = fruits.find(fruit => fruit.startsWith("a")); const index = fruits.indexOf("banana"); const hasApple = fruits.includes("apple"); console.log("倍数: ", doubled); console.log("偶数: ", evens); console.log("合計: ", sum);

実践プロジェクト

// タスク管理アプリの例 class TaskManager { constructor() { this.tasks = []; this.nextId = 1; } addTask(title) { const task = { id: this.nextId++, title: title, completed: false, createdAt: new Date() }; this.tasks.push(task); return task; } completeTask(id) { const task = this.tasks.find(t => t.id === id); if (task) { task.completed = true; } } getTasks() { return this.tasks; } getCompletedTasks() { return this.tasks.filter(t => t.completed); } } // 使用例 const manager = new TaskManager(); manager.addTask("JavaScriptを学習する"); manager.addTask("プロジェクトを完成させる"); manager.completeTask(1);
